Exercício 7

#include <stdio.h>

int main()
{
    int vetor[3], *p[3], aux[3]={0,0,0};

    for(int i=0; i<3; i++){
        p[i] = &vetor[i];
    }
    for(int i=0; i<3; i++){
        printf("\nValor %d: ", i+1);
        scanf("%d", p[i]);
    }

    for(int i=0; i<3; i++){
        for(int j=i+1; j<3; j++){
            if(vetor[i]>vetor[j]){
                aux[j]++;
            }
        }
    }
    for(int i=2; i>=0; i--){
        for(int j=0; j<3; j++){
            if(aux[j]==i){
                printf("\nValor %d: %d - Endereco: %p", j+1, vetor[j], p[j]);
            }
        }
    }
    
    return 0;
}
