#include <stdlib.h>

int main()
{
    printf("for: \n");

    for(int i=1; i<=100; i++){
        printf("%d ", i);
    }

    printf("\n\nwhile: \n");

    int i=1;
    while(i<=100){
        printf("%d ", i);
        i++;
    }

    printf("\n\ndo-while: \n");

    i=1;
    do{
        printf("%d ", i);
        i++;
    }while(i<=100);

    return 0;
}

Exercício 16

#include <stdio.h>
#include <stdlib.h>

 int main()
{
    int opcao;
    float x, y;

    do{
        printf("======= Menu de opcoes =======");
        printf("\n1-        Adicao(x+y)");
        printf("\n2-      Subtracao (x-y)");
        printf("\n3-    Multiplicacao (xy)");
        printf("\n4-       Divisao (x/y)");
        printf("\n5-           Saída");
        printf("\nOpcao: ");
        scanf("%d", &opcao);

        if((opcao>0)&&(opcao<5)){
            printf("\nx = ");
            scanf("%f", &x);
            printf("y = ");
            scanf("%f", &y);
        }

        switch(opcao){
            case 1:
                printf("\n%.2f+%.2f = %.2f\n\n", x, y, x+y);
                break;
            case 2:
                printf("\n%.2f-%.2f = %.2f\n\n", x, y, x-y);
                break;
            case 3:
                printf("\n%.2f%.2f = %.2f\n\n", x, y, x*y);
                break;
            case 4:
                if(y==0){
                    printf("\nValor invalido para y!\n\n");
                }
                else{
                    printf("\n%.2f/%.2f = %.2f\n\n", x, y, x/y);
                }
                break;
            default:
                printf("\nOpcao invalida!\n\n");
                opcao = 1;
                break;
        }

    }while((opcao>0)&&(opcao<5));

    return 0;
}
